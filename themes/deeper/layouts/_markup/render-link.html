{{- /* Links: external (http/https) open in a new tab; mailto:/tel: pass through (safeURL keeps tel: from #ZgotmplZ); pure #anchors pass through; internal /paths are resolved against the content tree with site.GetPage and rewritten to the canonical .RelPermalink (so links survive URL/slug changes and .html→ pretty). A dead internal link fails the build (params.linkcheck = "error") or warns (= "warn"). GetPage takes a CONTENT path, not an output URL — write internal links by content path (reference/features-recipes.md §1b). */ -}} {{- $d := .Destination -}} {{- $u := urls.Parse $d -}} {{- $scheme := $u.Scheme -}} {{- if $scheme -}} {{- if or (eq $scheme "http") (eq $scheme "https") -}} {{ .Text | safeHTML }} {{- else -}} {{ .Text | safeHTML }} {{- end -}} {{- else if hasPrefix $d "#" -}} {{ .Text | safeHTML }} {{- else if and (hasPrefix $d "/") (path.Ext $u.Path) -}} {{- /* link to a generated/static file (RSS /index.xml, sitemap, an image or a download): pass through as-is — GetPage resolves content pages, not output URLs, so these must not go through the page check. */ -}} {{ .Text | safeHTML }} {{- else if hasPrefix $d "/" -}} {{- $target := site.GetPage $u.Path -}} {{- if $target -}} {{ .Text | safeHTML }} {{- else -}} {{- $msg := printf "broken internal link %q on page %q" $d .Page.Path -}} {{- if eq (site.Params.linkcheck | default "error") "warn" -}}{{ warnf "%s" $msg }}{{- else -}}{{ errorf "%s" $msg }}{{- end -}} {{ .Text | safeHTML }} {{- end -}} {{- else -}} {{ .Text | safeHTML }} {{- end -}}
