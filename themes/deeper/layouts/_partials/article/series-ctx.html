{{- /* Series context for the current article, resolved once and shared by kicker, rail-left and series-bridge. Returns a dict: isSeries bool — true only when the term has >1 page term page — the series term label string — short lowercase series name (params.label | lower title) href string — rubric anchor, e.g. /internals/#memory slug string — series slug (= anchor id) pages pages — series parts ordered by series_weight pos,total int — 1-based position of this page, and part count A non-series page returns {isSeries:false}. */ -}} {{- $ctx := dict "isSeries" false -}} {{- with .GetTerms "series" -}} {{- $term := index . 0 -}} {{- $pages := partialCached "series/pages.html" $term $term.Path -}} {{- $pos := 0 -}} {{- range $i, $p := $pages }}{{ if eq $p.RelPermalink $.RelPermalink }}{{ $pos = add $i 1 }}{{ end }}{{ end -}} {{- $slug := $term.File.ContentBaseName -}} {{- $ctx = dict "isSeries" (gt (len $pages) 1) "term" $term "label" ($term.Params.label | default (lower $term.Title)) "slug" $slug "href" (printf "%s#%s" $.CurrentSection.RelPermalink $slug) "pages" $pages "pos" $pos "total" (len $pages) -}} {{- end -}} {{- return $ctx -}}
