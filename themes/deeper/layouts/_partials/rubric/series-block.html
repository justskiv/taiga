{{- /* Series sections of a rubric page. One
per series that has parts in this rubric, ordered by the series term's own weight. The terms are reached through each page's .GetTerms "series" (render:never terms aren't returned by site.GetPage, but .GetTerms resolves them), deduped by slug. Part count and duration sum the whole series (stubs included) so they match the article kicker's "часть N из M" and the series bridge. Context = the section page; part title uses .LinkTitle so a part can carry a short title (linkTitle:) without touching the guide's . */ -}} {{- $section := . -}} {{- $seen := dict -}} {{- $blocks := slice -}} {{- range $section.RegularPages -}} {{- with .GetTerms "series" -}} {{- $term := index . 0 -}} {{- $slug := $term.File.ContentBaseName -}} {{- if not (isset $seen $slug) -}} {{- $seen = merge $seen (dict $slug true) -}} {{- $pages := where $term.Pages "Section" $section.Section -}} {{- $blocks = $blocks | append (dict "term" $term "pages" $pages "slug" $slug "w" $term.Weight) -}} {{- end -}} {{- end -}} {{- end -}} {{- range sort $blocks "w" -}} {{- $term := .term -}} {{- $pages := .pages -}} {{- $mins := 0 -}}{{ range $pages }}{{ $mins = add $mins (int (.Params.mins | default .ReadingTime)) }}{{ end -}} <section class="series" id="{{ .slug }}"> <div class="ser-head"> <h2 class="ser-t">{{ $term.Title }}</h2> <span class="ser-k">{{ i18n "series_word" }}{{ with $term.Params.tagline }} · {{ . }}{{ end }}</span> <span class="ser-m">{{ i18n "parts_count" (len $pages) }} · ~{{ partial "rubric/duration.html" $mins }}</span> </div> {{- with $term.Description }} <p class="ser-d">{{ . }}</p> {{- end }} <ol class="slist"> {{- range $pages }} <li><a href="{{ .RelPermalink }}">{{ .LinkTitle }} <span class="mins">{{ .Params.mins | default .ReadingTime }} {{ i18n "mins_abbr" }}</span></a></li> {{- end }} </ol> </section> {{ end -}}
