{{- /* Links: external (http/https) open in a new tab; mailto:/tel: pass through (safeURL keeps tel: from #ZgotmplZ); pure #anchors pass through; internal /paths are resolved against the content tree with site.GetPage and rewritten to the canonical .RelPermalink (so links survive URL/slug changes and .html→ pretty). A dead internal link fails the build (params.linkcheck = "error") or warns (= "warn"). GetPage takes a CONTENT path, not an output URL — write internal links by content path (reference/features-recipes.md §1b). */ -}} {{- $d := .Destination -}} {{- $u := urls.Parse $d -}} {{- $scheme := $u.Scheme -}} {{- if $scheme -}} {{- if or (eq $scheme "http") (eq $scheme "https") -}} {{ .Text | safeHTML }} {{- else -}} {{ .Text | safeHTML }} {{- end -}} {{- else if hasPrefix $d "#" -}} {{ .Text | safeHTML }} {{- else if and (hasPrefix $d "/") (path.Ext $u.Path) -}} {{- /* Link to a generated/static file (RSS /index.xml, sitemap, an image or a download): skip the page check — GetPage resolves content pages, not output URLs. But still send it through relURL, or a root-absolute path escapes the baseURL subpath: on a site served from /taiga/, a literal "/index.xml" points at the domain root and 404s. relURL (not relLangURL) on purpose — it rebases the path and nothing else, so a shared static asset does not sprout a language prefix it does not have on disk. Write the language-specific path yourself when you mean one. The leading slash is trimmed first, and that is not cosmetic: relURL leaves an already-rooted "/index.xml" alone and only rebases a relative "index.xml". Feed it the slash and it silently no-ops — which is exactly how this bug shipped. (footer.html writes "index.xml" for the same reason.) */ -}} {{ .Text | safeHTML }} {{- else if hasPrefix $d "/" -}} {{- $target := site.GetPage $u.Path -}} {{- if $target -}} {{ .Text | safeHTML }} {{- else -}} {{- $msg := printf "broken internal link %q on page %q" $d .Page.Path -}} {{- if eq (site.Params.linkcheck | default "error") "warn" -}}{{ warnf "%s" $msg }}{{- else -}}{{ errorf "%s" $msg }}{{- end -}} {{ .Text | safeHTML }} {{- end -}} {{- else -}} {{ .Text | safeHTML }} {{- end -}}
