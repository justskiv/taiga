{{- /* Rubric showcase on the home page. Cards are generated per site section (site.Params.rubricSections, in listed order) — not a hard-coded three — so a site with more or fewer rubrics just works. The section logo is a partial looked up by name: home/logos/
.html on the SITE side (the mock's three SVGs live there, keeping the theme subject-agnostic), falling back to the theme's neutral home/logos/default.html. .r-n counts the section's whole library (series parts included) so it matches what the rubric page lists; when the section carries series it reads "N guides · M series", otherwise "N deep dives". params.home.rubricCards = "naked" drops the card box (fill + border) and grows the logo — free columns on the page, the vdoing mechanic; the default "boxed" is today's card, untouched. */ -}} {{- $cards := (site.Params.home | default dict).rubricCards | default "boxed" -}}
{{- range site.Params.rubricSections }} {{- $sec := site.GetPage (printf "/%s" .) -}} {{- if $sec }} {{- $id := $sec.Section -}} {{- $pages := where site.RegularPages "Section" $id -}} {{- $nGuides := len $pages -}} {{- /* Series are the rubric's sub-sections minus standalone containers; announced-but-empty series count too — they are already visible as teasers on the rubric page. */ -}} {{- $nSeries := 0 -}} {{- range $sec.Sections }}{{ if not .Params.standalone }}{{ $nSeries = add $nSeries 1 }}{{ end }}{{ end -}} {{- $logo := printf "home/logos/%s.html" $id -}} {{- $logoPartial := cond (templates.Exists (printf "_partials/%s" $logo)) $logo "home/logos/default.html" -}} {{ partial $logoPartial $sec }} {{ $sec.Title }} {{ with $sec.Params.card_line }}{{ . }}{{ end }} {{/* Count line only when the rubric has guides (with series → "N guides · M series", else "N deep dives"); an empty rubric shows a muted "in the works" rather than a meaningless "0 deep dives". */}} {{ if gt $nGuides 0 }}{{ if gt $nSeries 0 }}{{ i18n "guides_count" $nGuides }} · {{ i18n "series_count" $nSeries }}{{ else }}{{ i18n "reviews_count" $nGuides }}{{ end }}{{ else }}{{ i18n "rubric_empty" }}{{ end }} {{- end }} {{- end }}
