{{- /* Generate palette CSS from data/themes/.toml (theme + site merged): :root = the default palette (params.defaultTheme), plus one [data-theme] block per palette. Returned as a string; css.html prepends it to the bundle so the vars are defined before any rule consumes them. Palettes with `disabled = true` are skipped (a site can disable a theme's palette by overriding its file). Meta keys are not emitted as CSS vars. */ -}} {{- $skip := slice "name" "weight" "light" "disabled" -}} {{- $default := site.Params.defaultTheme | default "taiga" -}} {{- $themes := site.Data.themes -}} {{- /* Optional cross-cutting accent override (params.accent). When set, it replaces --accent/-dim/-glow in :root AND in every [data-theme] block, so the whole palette set adopts the site's brand colour while palettes stay data (ARCH §5a). Unset → empty dict → no substitution → output is byte-identical to the palette files. See docs/customizing.md. */ -}} {{- $ov := dict -}} {{- with site.Params.accent -}} {{- if findRE `^#[0-9a-fA-F]{6}$` . -}} {{- $r := int (printf "0x%s" (substr . 1 2)) -}} {{- $g := int (printf "0x%s" (substr . 3 2)) -}} {{- $b := int (printf "0x%s" (substr . 5 2)) -}} {{- $ov = dict "accent" . "accent-dim" (site.Params.accentDim | default (printf "rgba(%d,%d,%d,.18)" $r $g $b)) "accent-glow" (site.Params.accentGlow | default (printf "rgba(%d,%d,%d,.28)" $r $g $b)) -}} {{- else -}} {{- errorf "params.accent must be #rrggbb (6 hex digits), got %q" . -}} {{- end -}} {{- end -}} {{- $css := "" -}} {{- /* Light palettes opt out of the brand-accent override: --accent doubles as the CTA *background* under near-white --bg-deep text, so a bright brand accent fails contrast there. A light palette keeps its own (dark) accent from its data file. See light.toml. */ -}} {{- $root := index $themes $default -}} {{- with $root -}} {{- $rov := $ov -}}{{- if .light }}{{ $rov = dict }}{{ end -}} {{- $vars := "" -}} {{- range $k, $v := . -}}{{- if not (in $skip $k) -}}{{- with index $rov $k }}{{ $v = . }}{{ end -}}{{- $vars = printf "%s--%s:%s;" $vars $k $v -}}{{- end -}}{{- end -}} {{- $css = printf ":root{%s}" $vars -}} {{- end -}} {{- range $id, $t := $themes -}} {{- if not $t.disabled -}} {{- $pov := $ov -}}{{- if $t.light }}{{ $pov = dict }}{{ end -}} {{- $vars := "" -}} {{- range $k, $v := $t -}}{{- if not (in $skip $k) -}}{{- with index $pov $k }}{{ $v = . }}{{ end -}}{{- $vars = printf "%s--%s:%s;" $vars $k $v -}}{{- end -}}{{- end -}} {{- $css = printf "%s\n[data-theme=%q]{%s}" $css $id $vars -}} {{- end -}} {{- end -}} {{- return $css -}}
