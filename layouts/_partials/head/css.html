{{- /* CSS bundle: generated palette vars (first) + the numbered theme sheets in name order (the cascade) + the site's optional css/custom.css (last, so site overrides win without a fork). dev serves it raw; prod minifies and fingerprints. */ -}} {{- $palettes := partial "head/palettes.html" . | resources.FromString "css/00-palettes.css" -}} {{- $bundle := slice $palettes -}} {{- range sort (resources.Match "css/[0-9]*.css") "Name" -}} {{- $bundle = $bundle | append . -}} {{- end -}} {{- with resources.Get "css/custom.css" -}}{{ $bundle = $bundle | append . }}{{- end -}} {{- $css := $bundle | resources.Concat "css/bundle.css" -}} {{- if hugo.IsProduction -}} {{- with $css | minify | fingerprint "sha256" }} {{- end -}} {{- else -}} {{- end -}} {{- /* page-bundle widget styles: one folder per widget under widgets/, plus a _shared/ level for classes several widgets use. Concatenated in name order (_shared sorts before w-*), minified + fingerprinted, linked only here. The Concat target MUST be page-unique: Hugo caches concatenated resources by target path, so a fixed name would collide across pages — every page would serve whichever bundle won the cache. Key it by the page's UniqueID. */ -}} {{- $wcss := sort (.Resources.Match "widgets/**.css") "Name" -}} {{- with $wcss -}} {{- $bundle := . | resources.Concat (printf "widgets/bundle-%s.css" $.File.UniqueID) -}} {{- if hugo.IsProduction -}} {{- with $bundle | minify | fingerprint "sha256" }} {{- end -}} {{- else -}} {{- end -}} {{- end -}}
